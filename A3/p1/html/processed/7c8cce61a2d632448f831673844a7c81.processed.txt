            function getDSTstart(){
                d = new Date(2016, 2, 1, 0, 0, 0, 0);
                var day = 0;
                if (d.getDay() != 0) {
                day = 8-(d.getDay())
                day = day + 7
                d = d.setDate(day)
                DST_start = new Date(d)
                }
            }
            function getDSTend(){
                d = new Date(2016, 10, 1, 0, 0, 0, 0);
                var day = 0;
                if (d.getDay() != 0) {
                day = 8-(d.getDay())
                d = d.setDate(day)
                DST_end = new Date(d)
                }
            }
            function setFooterTime(){
             getDSTstart();
             getDSTend();
            //EST
            clientDate = new Date();
            if(clientDate >= DST_start && clientDate <=DST_end){
                offset = -4;
            }else{
                offset = -5;
            }
            
            utc = clientDate.getTime() + (clientDate.getTimezoneOffset() * 60000);
            serverDate = new Date(utc + (3600000*offset));
            var dt = serverDate
            var hours = dt.getHours();
            //it is pm if hours from 12 onwards
            suffix = (hours >= 12)? 'p.m.' : 'a.m.';
            //only -12 from hours if it is greater than 12 (if not back at mid night)
            hours = (hours > 12)? hours -12 : hours;
            //if 00 then it is 12 am
            hours = (hours == '00')? 12 : hours;
            var minutes = dt.getMinutes();
            minutes = (minutes < 10)? '0'+minutes : minutes;
            var time = hours+':'+minutes+' '+suffix
            $('.realTime').text(time)
            $('#copy_year').html(dt.getFullYear());
            }
