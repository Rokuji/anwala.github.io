                            function initializeClock($elem) {
                                var $timer = $elem.find('.js-timer'),
                                    template = $timer.html(),
                                    startTime = new Date(Date.parse($elem.data('spoofDateStart'))),
                                    endTime = new Date(Date.parse($elem.data('dateTimeEnd'))),
                                    timeEndData = $elem.data('timeEnd');

									timeNextHour.setMinutes(0);
									timeNextHour.setSeconds(0);

									return timeNextHour;
								}

                                function updateClock($timer) {
                                    //Set start date current time if DW site preview is not being used to spoof date and time
                                    if (window.location.href.indexOf('__siteDate=') === -1) {
                                        startTime = Date.parse(new Date());

                                        $elem.addClass('hide');
                                    }
                                }

                                updateClock($timer);
                                var timeinterval = setInterval(updateClock, 1000, $timer);
                            }

                            function getTimeRemaining(startTime, endTime) {
                                var t = Date.parse(endTime) - startTime,
                                    seconds = Math.floor((t / 1000) % 60),
                                    minutes = Math.floor((t / 1000 / 60) % 60),
                                    hours = Math.floor((t / (1000 * 60 * 60)) % 24),
                                    days = Math.floor(t / (1000 * 60 * 60 * 24));
                                return {
                                    'total': t,
                                    'days': days,
                                    'hours': ('0' + hours).slice(-2),
                                    'minutes': ('0' + minutes).slice(-2),
                                    'seconds': ('0' + seconds).slice(-2)
                                };
                            }

                                    targetTime.setDate(targetTime.getDate() + 1);
                                }

                                return targetTime;
